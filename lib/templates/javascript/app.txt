Rally.createApp('@@NAME@@', {
	launch: function launch() {
		Rally.data.ModelFactory.getModel({
			type: 'UserStory',
			scope: this,
			success: function(model) {
				this.grid = this.add({
					xtype: 'ruigrid',
					model: model,
					columnCfgs: [
						'FormattedID',
						'Name',
						'Owner'
					],
					storeConfig: {
						filters: [
							{
								property: 'ScheduleState',
								operator: '=',
								value: 'Accepted'
							}
						]
					}
				});
			}
		});
	}
});
